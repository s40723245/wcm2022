Tetris

keycode

新增按下X鍵順時針旋轉

    def rotate1(self):
        self.rotation = (self.rotation - 1) % len(self.figures[self.type])
    def rotate1(self):
        old_rotation = self.figure.rotation
        self.figure.rotate1()
        if self.intersects():
            self.figure.rotation = old_rotation
    # 88 is x key to clockwise rotate
    if key == 88:
        game.rotate1()

修改按下Z鍵逆時針旋轉

    # 90 is z key to anticlockwise rotate
    if key == 90:
        game.rotate()

修改按下空白鍵讓方塊下降到最底下

    # 32 is space key to move block to bottom
    if key == 32:

    def go_down(self):
        self.figure.y += 1
        if self.intersects():
            self.figure.y -= 1
            self.freeze()
    # 67 is c key to drop -1 on the y-axis
    if key == 67:
        game.go_down()
